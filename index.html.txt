function getDays(start) {
  if (!start) return 0;
  return Math.floor((Date.now() - new Date(start).getTime()) / 86400000);
}
function nowStr() {
  return new Date().toLocaleString("es-EC", { day: "numeric", month: "short", hour: "2-digit", minute: "2-digit" });

function Ring({ days }) {
  const pct = Math.min(days / Math.max(days, 365), 1);
  const r = 65, circ = 2 * Math.PI * r;
  return (
    <svg width="165" height="165" style={{ transform: "rotate(-90deg)" }}>
      <circle cx="82.5" cy="82.5" r={r} fill="none" stroke="rgba(255,255,255,0.06)" strokeWidth="10" />
      <circle cx="82.5" cy="82.5" r={r} fill="none" stroke="url(#rg)" strokeWidth="10"
        strokeDasharray={`${pct * circ} ${circ}`} strokeLinecap="round"

          <stop offset="0%" stopColor="#95d5b2" />

}

                <p style={{ fontFamily: "'DM Sans',sans-serif", fontSize: 12, color: "#74c69d", marginBottom: 2 }}>📖 {h.query.slice(0,55)}{h.query.length > 55 ? "..." : ""}</p>
                <p style={{ fontFamily: "'DM Sans',sans-serif", fontSize: 10, color: "#3a6b4a" }}>{h.ts}</p>
              </div>
            ))}
          </>}
        </>}

      </div>

      {/* Tab Bar */}
      <div className="tab-bar">
        {tabs.map(t => (
          <button key={t.id} className={`tb ${tab === t.id ? "on" : ""}`} onClick={() => setTab(t.id)}>
            <span>{t.icon}</span><span>{t.label}</span>
          </button>
        ))}
      </div>
    </div>
  );
}
